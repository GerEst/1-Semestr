turtle.penup()

def drawing(n):
    x = 0
    for element in str(n):
        for i in NM[element]:
            for j in i:

                if  NM[element].index(i) == 0:      
                    turtle.penup()
                    turtle.goto(j[0]*50 + x,j[1]*50)
                else:
                    turtle.pendown()
                    turtle.goto(j[0]*50 + x,j[1]*50)

        x += 100
