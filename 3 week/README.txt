- добавление\выбрасывание элемента списка
    guests.remove("Лида и Наташа")
    print(guests)  # ['Родители', 'Одноклассники']
    guests.append('Дима')
    guests.append('Света')
    print(guests)  # ['Родители', 'Одноклассники', 'Дима', 'Света']

- изменение элемента по его индексу
    guests = ['Родители', 'Одноклассники', 'Дима', 'Света']
    print(guests)  # ['Родители', 'Одноклассники', 'Дима', 'Света']
    guests[2] = 'Коллеги'
    print(guests)  # ['Родители', 'Одноклассники', 'Коллеги', 'Света']

- элементы также доступны по индексу
    books = ("1984", "О дивный новый мир", "451 градус по Фаренгейту")
    print(books[2]) # 451 градус по Фаренгейту

- узнать есть ли данное в кортеже (True\False)
    books = ("1984", "О дивный новый мир", "451 градус по Фаренгейту")
    print("1984" in books)  # True
    print("Незнайка на луне" in books)  # False

СЛОВАРЬ = {}

Используются для связывания одного объекта, называемого ключом, с другим, называемым значением. = ОТОБРАЖЕНИЕ
Вы можете найти в словаре ключ и получить соответствующее ему значение

- Словари, как и списки, изменяемы. То есть, в них можно добавлять новые пары ключ-значение.

    phones = {
    "Иван": "+7123456789",
    "Василий": "+37520123456"
    }
      print(phones)

- Для определения того, есть ли ключ в словаре, используйте ключевое слово in

    print("Василий" in phones)  # True

- Можно удалить пару ключ-значение с помощью ключевого слова del.

    del phones["Василий"]

- сохранять контейнеры в других контейнерах. Например, можно хранить списки внутри списка.

    music = {
        "rap": ["Баста", "Кравц", "Злой Дух"],
        "rock": ["Наутилус Помпилиус", "Кино", "Ария"],
        "djs": ["Paul Oakenfold", "Tiesto"]
    }
    print(music['rap'])  # ['Баста', 'Кравц', 'Злой Дух']
    print(music['rock'][-1])  # Ария
